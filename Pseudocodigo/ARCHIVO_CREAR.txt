Accion REGISTRO ES;

Ambiente
	fecha = REGISTRO
			dia: 1..31;
			mes: 1..12;
			año: 1900..9999;
	fin_registro;	

	persona = REGISTRO
		nombre: AN(50);
		dir: AN(50);
		mail: AN(80);
		fecha_nac: fecha;
	fin_registro;

	arch: ARCHIVO de persona;
	reg: persona;

	opcion: caracer;
Algoritmo
	
	ABRIRs(arch);

	opcion := "S"

	MIENTRAS opcion = "S" HACER

		ESCRIBIR('Ingrese los datos de la persona...');

		ESCRIBIR('Ingrese Nombre...');		
		LEER(reg.nombre)

		ESCRIBIR('Ingrese Domicilio...');		
		LEER(reg.dir)

		ESCRIBIR('Ingrese E-Mail...');		
		LEER(reg.mail)

		ESCRIBIR('Ingrese Fecha de Nacimiento...');		
		ESCRIBIR('Ingrese Dia...');		
		LEER(reg.fecha_nac.dia)

		ESCRIBIR('Ingrese mes...');		
		LEER(reg.fecha_nac.mes)

		ESCRIBIR('Ingrese Año...');		
		LEER(reg.fecha_nac.año)

		ESCRIBIR(arch, reg);

		ESCRIBIR('DESEA CONTINUAR CARGANDO REGISTROS? S/N..');
		LEER(opcion);
	FinMientras;

	CERRAR(arch);
FinAccion.
